2. DEPARTMENT & POLICE HIERARCHY
CREATE TABLE Departments (
    department_id      NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    department_name    VARCHAR2(200) NOT NULL,
    department_code    VARCHAR2(50) UNIQUE,
    parent_department_id NUMBER,
    location           VARCHAR2(200),
    status             VARCHAR2(20) DEFAULT 'ACTIVE',
    created_at         TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at         TIMESTAMP,

    CONSTRAINT fk_parent_department
    FOREIGN KEY (parent_department_id)
    REFERENCES police_departments(department_id)
);

CREATE TABLE police_ranks (
    rank_id        NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    rank_name      VARCHAR2(100) NOT NULL,
    rank_code      VARCHAR2(50) UNIQUE,
    hierarchy_level NUMBER,
    description    VARCHAR2(255),
    status         VARCHAR2(20) DEFAULT 'ACTIVE'
);

5. COMPLAINT/Petition REGISTRATION

CREATE TABLE complaint (
    complaint_id        NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    complaint_number    VARCHAR2(50) UNIQUE NOT NULL,

    user_id             NUMBER DEFAULT 99999,
    is_anonymous        CHAR(1) DEFAULT 'N',

    complaint_mode      VARCHAR2(30),
    complaint_type      VARCHAR2(50),

    title               VARCHAR2(255),
    description         CLOB,

    incident_datetime   TIMESTAMP,
    reported_datetime   TIMESTAMP DEFAULT SYSTIMESTAMP,

    location_text       VARCHAR2(500),
    latitude            NUMBER(10,7),
    longitude           NUMBER(10,7),

    district            VARCHAR2(100),
    state               VARCHAR2(100),

    status              VARCHAR2(30) DEFAULT 'REGISTERED',

    created_at          TIMESTAMP DEFAULT SYSTIMESTAMP,
    updated_at          TIMESTAMP,

    CONSTRAINT fk_complaint_user FOREIGN KEY (user_id)
        REFERENCES users(user_id)
);
