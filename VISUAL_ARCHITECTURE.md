â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper Functions
    function isPolice() {
      return request.auth != null && 
             request.auth.token.role == 'police';
    }
    
    function getPoliceRank() {
      return request.auth.token.rank;
    }
    
    function getPoliceStation() {
      return request.auth.token.stationName;
    }
    
    function getPoliceDistrict() {
      return request.auth.token.district;
    }
    
    function isStateLevelRank() {
      return getPoliceRank() in [
        'Director General of Police',
        'Additional Director General of Police'
      ];
    }
    
    function isRangeLevelRank() {
      return getPoliceRank() in [
        'Inspector General of Police',
        'Deputy Inspector General of Police'
      ];
    }
    
    function isDistrictLevelRank() {
      return getPoliceRank() in [
        'Superintendent of Police',
        'Additional Superintendent of Police'
      ];
    }
    
    function isStationLevelRank() {
      return getPoliceRank() in [
        'Deputy Superintendent of Police',
        'Inspector of Police',
        'Sub Inspector of Police',
        'Assistant Sub Inspector of Police',
        'Head Constable',
        'Police Constable'
      ];
    }
    
    // Petition Access Rules
    match /petitions/{petitionId} {
      allow read: if isPolice() && (
        // State level: Can view all
        isStateLevelRank() ||
        
        // Range level: Can view from their range (needs range field)
        (isRangeLevelRank() && 
         resource.data.range == request.auth.token.range) ||
        
        // District level: Can view from their district
        (isDistrictLevelRank() && 
         resource.data.district == getPoliceDistrict()) ||
        
        // Station level: Can view from their station ONLY
        (isStationLevelRank() && 
         resource.data.stationName == getPoliceStation())
      );
      
      allow update: if isPolice(); // Can update status
    }
    
    // Police Profile Access
    match /police/{userId} {
      allow read: if isPolice();
      allow write: if request.auth.uid == userId;
    }
  }
}
